for i in node_t:

                                     vtype=GRB.INTEGER,
                                     name="h" + '_' + i + '_' + j)

for i in node_t:
    obj = LinExpr(0)
    for j in h.keys():
        if j[0] == i:
            obj.addTerms(1, h[j])

            obj.addTerms(-1, h[j])
    if i == 'O0':
        m.addConstr(obj == q)

UB = 11111111111110
LB = -11111111110
while UB -LB>1:
    H = {}
    for i in h.keys():
            H[h[i].VarName] = h[i].x
    constraint_all = 0
    objectiv = 0
    for case in cases:
        constraint = 0
        model = Model("Subproblem")
        z = {}
        s = {}

        for i in t_interval:
            for a in node:
                for b in node:
                    z[a, b, i] = model.addVar(lb=0,
                                                ub=100,
                                                vtype=GRB.CONTINUOUS,
                                                name="z" + '_' + a + '_' + b +
                                                '_' + str(i))

        for i in node:
            for a in t_interval:
                key = i + str(a)
                key1 = i + str(a + 1)
                obj = LinExpr(0)
                for j in node:
                    obj.addTerms(1, z[j, i, a])
                model.addConstr(obj <= Capacity * H["h" + '_' + key + '_' + key1],name="h"+i+str(a))

        for i in node:
            for a in t_interval:
                s[i, a] = model.addVar(lb=0,
                                         ub=100,
                                         vtype=GRB.CONTINUOUS,
                                         name="s" + '_' + i + '_' + str(a))

        for i in node:
            for a in t_interval:
                obj = LinExpr(0)
                key = i + str(a)
                for j in node:
                    obj.addTerms(1, z[i, j, a])
                model.addConstr(obj + s[i, a] >= demand[case][key])
        obj = 0
        obj1 = LinExpr(0)
        aaa = 0
        for i in t_interval:
            for a in node:
                obj1.addTerms(30, s[a, i])
                for b in node:

                    if a == b:
                        obj += 0 * z[a, b, i]
                    else:
                        obj += short_path[str(i)] * z[a, b, i]
                    aaa += 1
        model.setObjective(obj + obj1, GRB.MINIMIZE)
        model.Params.OutputFlag = 0
        model.optimize()  
        pi = {}
        for con in model.getConstrs():
            pi[con.ConstrName] = con.Pi
        for i in node:
            for a in t_interval:
                key = i + str(a)
                key1 = i + str(a + 1)
                obj = 0
                for j in node:
                    obj += z[j, i, a].x
                name = "h" + i + str(a)
                constraint += -pi[name] * (obj - Capacity * h[key, key1])
        constraint += model.objVal
        objectiv += 0.333*model.objVal
        constraint = 0.333*constraint
        constraint_all += constraint
    UB = min(UB,objectiv+3000*q.x)
    m.addConstr(constraint_all <= theta)
    m.Params.OutputFlag = 0
    m.write("Master1.lp")
    m.optimize()
    LB = m.ObjVal
    print('UB:',UB)
    print('LB:',LB)
